void AppsMenu::drawIcon(float scale) {
    clearIconArea();

    int iconW = scale * 60;
    int iconH = scale * 60;
    if (iconW % 2 != 0) iconW++;
    if (iconH % 2 != 0) iconH++;

    int cx = iconCenterX;
    int cy = iconCenterY;
    int half = iconW / 2;

    tft.drawRoundRect(cx - half, cy - half, iconW, iconH, half / 4, bruceConfig.priColor);
    tft.drawLine(cx - half, cy - half + iconH / 4, cx + half, cy - half + iconH / 4, bruceConfig.priColor);

    int gridW = iconW / 5;
    int gridGap = iconW / 8;

    int gridCenterY = cy + iconH / 8;
    int gx1 = cx - gridW - gridGap / 2;
    int gy1 = gridCenterY - gridW - gridGap / 2;
    int gx2 = cx + gridGap / 2;
    int gy2 = gridCenterY + gridGap / 2;

    tft.fillRect(gx1, gy1, gridW, gridW, bruceConfig.priColor);
    tft.drawRect(gx2, gy1, gridW, gridW, bruceConfig.priColor);
    tft.drawRect(gx1, gy2, gridW, gridW, bruceConfig.priColor);
    tft.fillRect(gx2, gy2, gridW, gridW, bruceConfig.priColor);
}

========== EthernetMenu.cpp ==========
void EthernetMenu::drawIcon(float scale) {
    clearIconArea();

    int iconW = scale * 50;
    int iconH = scale * 50;
    int cx = iconCenterX;
    int cy = iconCenterY;

    int halfW = iconW / 2;
    int halfH = iconH / 2;

    int r = scale * 4;

    // Outer thick connector frame
    int outX = cx - halfW;
    int outY = cy - halfH;

    // Draw main frame
    tft.drawRoundRect(outX, outY, iconW, iconH, r, bruceConfig.priColor);
    tft.drawRoundRect(outX + 1, outY + 1, iconW - 2, iconH - 2, r - 1, bruceConfig.priColor);

    // Draw the notch (the clip hole of RJ45) at the bottom
    int notchW = iconW / 2.2;
    int notchH = iconH / 3.5;
    int notchX = cx - notchW / 2;
    int notchY = outY + iconH - notchH;

    tft.fillRect(notchX, notchY - 1, notchW, notchH + 2, bruceConfig.bgColor);

    // Draw the indentation lines for the notch
    tft.drawLine(notchX, notchY, notchX, notchY + notchH, bruceConfig.priColor);
    tft.drawLine(notchX + notchW, notchY, notchX + notchW, notchY + notchH, bruceConfig.priColor);
    tft.drawLine(notchX, notchY, cx - halfW, notchY, bruceConfig.priColor);
    tft.drawLine(notchX + notchW, notchY, cx + halfW, notchY, bruceConfig.priColor);

    // Draw pins at the top
    int pinW = scale * 2;
    if (pinW < 1) pinW = 1;
    int pinH = iconH / 3;
    int pinY = outY + 3;
    int numPins = 6;
    int space = (iconW - 14 * scale) / numPins;

    int startX = cx - (space * (numPins - 1)) / 2;
    for (int i = 0; i < numPins; i++) {
        tft.fillRect(startX + i * space - pinW / 2, pinY, pinW, pinH, bruceConfig.priColor);
    }
}

void ScriptsMenu::drawIcon(float scale) {
    clearIconArea();

    int iconW = scale * 40;
    int iconH = scale * 60;

    if (iconW % 2 != 0) iconW++;
    if (iconH % 2 != 0) iconH++;

    int foldSize = iconH / 4;
    int arrowSize = iconW / 10;
    int arrowPadX = 2 * arrowSize;
    int arrowPadBottom = 3 * arrowPadX;
    int slashSize = 2 * arrowSize;

    // File
    tft.drawRect(iconCenterX - iconW / 2, iconCenterY - iconH / 2, iconW, iconH, bruceConfig.priColor);
    tft.fillRect(
        iconCenterX + iconW / 2 - foldSize, iconCenterY - iconH / 2, foldSize, foldSize, bruceConfig.bgColor
    );
    tft.drawTriangle(
        (iconCenterX + iconW / 2 - foldSize),
        (iconCenterY - iconH / 2),
        (iconCenterX + iconW / 2 - foldSize),
        (iconCenterY - iconH / 2 + foldSize - 1),
        (iconCenterX + iconW / 2 - 1),
        (iconCenterY - iconH / 2 + foldSize - 1),
        bruceConfig.priColor
    );

    // Left Arrow
    tft.drawLine(
        iconCenterX - iconW / 2 + arrowPadX,
        iconCenterY + iconH / 2 - arrowPadBottom,
        iconCenterX - iconW / 2 + arrowPadX + arrowSize,
        iconCenterY + iconH / 2 - arrowPadBottom + arrowSize,
        bruceConfig.priColor
    );
    tft.drawLine(
        iconCenterX - iconW / 2 + arrowPadX,
        iconCenterY + iconH / 2 - arrowPadBottom,
        iconCenterX - iconW / 2 + arrowPadX + arrowSize,
        iconCenterY + iconH / 2 - arrowPadBottom - arrowSize,
        bruceConfig.priColor
    );

    // Slash
    tft.drawLine(
        iconCenterX - slashSize / 2,
        iconCenterY + iconH / 2 - arrowPadBottom + arrowSize,
        iconCenterX + slashSize / 2,
        iconCenterY + iconH / 2 - arrowPadBottom - arrowSize,
        bruceConfig.priColor
    );

    // Right Arrow
    tft.drawLine(
        iconCenterX + iconW / 2 - arrowPadX,
        iconCenterY + iconH / 2 - arrowPadBottom,
        iconCenterX + iconW / 2 - arrowPadX - arrowSize,
        iconCenterY + iconH / 2 - arrowPadBottom + arrowSize,
        bruceConfig.priColor
    );
    tft.drawLine(
        iconCenterX + iconW / 2 - arrowPadX,
        iconCenterY + iconH / 2 - arrowPadBottom,
        iconCenterX + iconW / 2 - arrowPadX - arrowSize,
        iconCenterY + iconH / 2 - arrowPadBottom - arrowSize,
        bruceConfig.priColor
    );
}
